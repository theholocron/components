# Bootstrap

A set of components to bootstrap an application.

## Installation

```bash
npm install --save-dev @theholocron/bootstrap
```

## Usage

```typescript
import { Bootstrap, type BootstrapProps } from "@theholocron/bootstrap";

function App (props: BootstrapProps) {
	return (
		<Bootstrap {...props} />
	);
}
```
